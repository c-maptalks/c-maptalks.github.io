DistanceTool

A map tool to help measure distance on the map

new DistanceTool(optionsopt) [source] 

var distanceTool = new DistanceTool({
    'once' : true,
    'symbol': {
      'lineColor' : '#34495e',
      'lineWidth' : 2
    },
    'vertexSymbol' : {
      'markerType'        : 'ellipse',
      'markerFill'        : '#1bbc9b',
      'markerLineColor'   : '#000',
      'markerLineWidth'   : 3,
      'markerWidth'       : 10,
     'markerHeight'      : 10
   },
   'language' : 'en-US'
 }).addTo(map);
